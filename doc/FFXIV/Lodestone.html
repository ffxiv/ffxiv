def fetch(endpoint)
  uri = "http://na.finalfantasyxiv.com/lodestone/" + endpoint
  d("Fetching #{uri} ......")
  html, charset = open(uri) do |page|
    [page.read, page.charset]
  end
  d("    done (size=#{html.size}, charset=#{charset})")
  Nokogiri::HTML.parse(html, nil, charset)
end
